🚀 한투 API 초당 호출 제한(EGW00201) 방어 로직 구현
한국투자증권 API는 초당 호출 유량을 엄격히 제한하고 있으며, 이를 초과할 경우 EGW00201 (초당 거래건수를 초과하였습니다) 에러를 반환하며 일시적으로 요청을 차단합니다
.
이를 방어하기 위해 1) 큐(Deque)를 이용해 1초 내에 발송된 요청 개수를 제어하는 RateLimiter 클래스와 2) 에러 발생 시 자동으로 1초 대기 후 재요청을 보내는 safe_api_call 래퍼 함수를 작성해야 합니다.
kis_api.py 파일에 아래 코드를 추가하고, 향후 모든 API 호출 시 requests.get 대신 safe_api_call을 사용하도록 합니다.
kis_api.py (쓰로틀링 로직 적용 버전)
import os
import time
import requests
import json
from collections import deque
from dotenv import load_dotenv

load_dotenv()

APP_KEY = os.getenv("KIS_MOCK_APP_KEY")
APP_SECRET = os.getenv("KIS_MOCK_APP_SECRET")
URL_BASE = "https://openapivts.koreainvestment.com:29443"

# ==========================================
# 1. 쓰로틀링(Rate Limiter) 클래스
# ==========================================
class RateLimiter:
    def __init__(self, max_calls, period=1.0):
        self.max_calls = max_calls
        self.period = period
        self.calls = deque()

    def wait_if_needed(self):
        """API 호출 전, 1초 내에 지정된 횟수 이상 요청이 있었는지 검사하고 대기합니다."""
        now = time.time()
        
        # 지정된 기간(1초)보다 오래된 호출 기록은 큐에서 제거
        while self.calls and now - self.calls > self.period:
            self.calls.popleft()
            
        # 만약 1초 내의 호출 횟수가 최대치를 꽉 채웠다면
        if len(self.calls) >= self.max_calls:
            # 가장 오래된 요청이 1초가 지날 때까지 강제로 대기 (Sleep)
            sleep_time = self.period - (now - self.calls)
            if sleep_time > 0:
                time.sleep(sleep_time)
                
        # 대기가 끝났거나 여유가 있다면 현재 시간을 기록하고 통과
        self.calls.append(time.time())

# 한투 API의 초당 호출 제한을 보수적으로 설정 (예: 초당 15회)
# 신규 가입 직후나 모의투자의 경우 간혹 더 낮게 제한되는 경우가 있으므로 조절 가능
kis_limiter = RateLimiter(max_calls=15, period=1.0)


# ==========================================
# 2. 안전한 API 호출 래퍼 (에러 발생 시 재시도)
# ==========================================
def safe_api_call(url, headers, data=None, method="GET", retries=3):
    """
    모든 한투 API 호출은 이 함수를 거칩니다.
    쓰로틀링을 적용하고, EGW00201 에러 발생 시 자동으로 재시도합니다.
    """
    for attempt in range(retries):
        # 1. 호출 전 유량 제한 확인 및 대기
        kis_limiter.wait_if_needed()
        
        # 2. 실제 API 호출
        if method == "POST":
            res = requests.post(url, headers=headers, data=json.dumps(data) if data else None)
        else:
            res = requests.get(url, headers=headers, params=data)
            
        res_data = res.json()
        
        # 3. 에러 코드 확인 (EGW00201: 초당 거래건수 초과)
        if res_data.get("msg_cd") == "EGW00201":
            print(f"[경고] 초당 호출 제한 도달 (EGW00201). 1.5초 대기 후 재시도... (Attempt {attempt+1}/{retries})")
            time.sleep(1.5)  # 넉넉하게 대기 후 다음 루프에서 재시도
            continue
            
        # 성공 시 데이터 반환
        return res_data
        
    # 최대 재시도 횟수 초과 시 에러 반환
    return {"error": "최대 재시도 횟수 초과", "last_response": res_data}


# ==========================================
# 3. 토큰 발급 함수 (safe_api_call 적용 예시)
# ==========================================
def get_access_token():
    headers = {"content-type": "application/json"}
    body = {
        "grant_type": "client_credentials",
        "appkey": APP_KEY,
        "appsecret": APP_SECRET
    }
    url = f"{URL_BASE}/oauth2/tokenP"
    
    # 일반 requests.post 대신 safe_api_call 사용
    res_data = safe_api_call(url, headers, data=body, method="POST")
    
    if "error" in res_data:
        return f"토큰 발급 실패: {res_data['error']}"
        
    return res_data.get("access_token", "토큰을 찾을 수 없습니다.")

💡 코드 구현 핵심 포인트
RateLimiter 큐잉 관리: 배열(deque)에 매 API 호출 시간을 기록해두고, 가장 오래된 호출 시간이 1초를 지나지 않았는데 15개가 꽉 찼다면 정확히 남은 시간만큼만 멈추고(time.sleep) 실행되도록 설계하여 병목을 최소화했습니다.
safe_api_call 함수 하나로 통합: 앞으로 잔고를 조회하든, 주식을 매수하든 무조건 이 함수를 통해 데이터를 보내도록 작성하세요. 개발자가 일일이 sleep()을 걸 필요 없이, 프로그램이 알아서 속도를 조절하고 초과 시 재시도(Retry) 해줍니다.
